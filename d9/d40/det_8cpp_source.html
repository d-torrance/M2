93 int DetComputation::step()
94 // Compute one more determinant of size p.
95 // increments I and/or J and updates 'dets', 'table'.
96 {
97  if (done) return COMP_DONE;
98 
99  ring_elem r;
100 
101  if (strategy == DET_BAREISS)
102  {
103  get_minor(row_set, col_set, p, D);
104  r = bareiss_det();
105  }
106  else
107  r = calc_det(row_set, col_set, p);
108 
109  if (!R->is_zero(r))
110  {
111  if (do_exterior)
112  result.set_entry(this_row, this_col, r);
113  else
114  result.append(R->make_vec(0, r));
115  }
116  else
117  R->remove(r);
118 
119  this_row++;
120  if (!Subsets::increment(M->n_rows(), p, row_set))
121  {
122  // Now we increment column
123  if (!Subsets::increment(M->n_cols(), p, col_set))
124  {
125  done = true;
126  return COMP_DONE;
127  }
128  // Now set the row set back to initial value
129  this_col++;
130  this_row = 0;
131  for (size_t i = 0; i < p; i++) row_set[i] = i;
132  }
133  return COMP_COMPUTING;
134 }

193  ring_elem g1,

229 ring_elem DetComputation::bareiss_det()
230 {
231  // Computes the determinant of the p by p matrix D. (dense form).
232  int sign = 1;
233  size_t pivot_col;
234 
235  ring_elem pivot = R->from_long(0);
236  ring_elem lastpivot = R->from_long(0);
237 
238  for (size_t r = p - 1; r >= 1; --r)
239  {
240  R->remove(lastpivot);
241  lastpivot = pivot;
242  if (!get_pivot(D, r, pivot, pivot_col)) // sets pivot_col and pivot
243  {
244  // Remove the rest of D.
245  for (size_t i = 0; i <= r; i++)
246  for (size_t j = 0; j <= r; j++) R->remove(D[i][j]);
247  R->remove(lastpivot);
248  return R->from_long(0);
249  }
250  for (size_t i = 0; i < r; i++) gauss(D, i, r, pivot_col, lastpivot);
251 
252  if (((r + pivot_col) % 2) == 1)
253  sign = -sign; // MES: do I need to rethink this logic?
254 
255  for (size_t c = 0; c <= r; c++)
256  if (c != pivot_col)
257  R->remove(D[r][c]);
258  else
259  D[r][c] = ZERO_RINGELEM;
260  }
261 
262  R->remove(pivot);
263  R->remove(lastpivot);
264  ring_elem r = D[0][0];
265  D[0][0] = ZERO_RINGELEM;
266 
267  if (sign < 0) R->negate_to(r);
268 
269  return r;
270 }
271 ring_elem DetComputation::calc_det(size_t *r, size_t *c, int p0)
272 // Compute the determinant of the minor with rows r[0]..r[p0-1]
273 // and columns c[0]..c[p0-1].
274 {
275  if (p0 == 1) return M->elem(static_cast<int>(r[0]), static_cast<int>(c[0]));
276  ring_elem answer = R->from_long(0);
277 
278  int negate = 1;
279  for (int i = p0 - 1; i >= 0; i--)
280  {
281  std::swap(c[i], c[p0 - 1]);
282  negate = !negate;
283  ring_elem g =
284  M->elem(static_cast<int>(r[p0 - 1]), static_cast<int>(c[p0 - 1]));
285  if (R->is_zero(g))
286  {
287  R->remove(g);
288  continue;
289  }
290  ring_elem h = calc_det(r, c, p0 - 1);
291  ring_elem gh = R->mult(g, h);
292  R->remove(g);
293  R->remove(h);
294  if (negate)
295  R->subtract_to(answer, gh);
296  else
297  R->add_to(answer, gh);
298  }
299 
300  // pulling out the columns has disordered c. Fix it.
301 
302  size_t temp = c[p0 - 1];
303  for (size_t i = p0 - 1; i > 0; i--) c[i] = c[i - 1];
304  c[0] = temp;
305 
306  return answer;
307 }
