472 bool gbB::pair_not_needed(spair *p, gbelem *m)
473 {
474  /* Check the criterion: in(m) divides lcm(p).
475  * If so: check if lcm(p1,m) == lcm(p) (if so, return false)
476  * check if lcm(p2,m) == lcm(p) (if so, return false)
477  * If still here, return true.
478  */
479  int i, first, second;
480  bool firstok;
481  exponents mexp, lcm, p1exp, p2exp;
482  if (p->type != SPAIR_SPAIR && p->type != SPAIR_RING) return false;
483  mexp = m->lead;
484  lcm = p->lcm;
485  if (gbelem_COMPONENT(m) != spair_COMPONENT(p)) return false;
486 
487  first = p->x.pair.i;
488  second = p->x.pair.j;
489  p1exp = gb[first]->lead;
490  p2exp =
491  gb[second]->lead; /* If a ring pair, this should index into gb array */
492 
493  ncalls++;
494  for (i = 0; i < nvars; i++, nloops++)
495  if (mexp[i] > lcm[i]) return false;
496 
497  firstok = false;
498  for (i = 0; i < nvars; i++)
499  {
500  if (mexp[i] == lcm[i]) continue;
501  if (p1exp[i] == lcm[i]) continue;
502  firstok = true;
503  break;
504  }
505  if (!firstok) return false;
506  for (i = 0; i < nvars; i++)
507  {
508  if (mexp[i] == lcm[i]) continue;
509  if (p2exp[i] == lcm[i]) continue;
510  return true;
511  }
512  return false;
513 }

1289 // h = f - sum(a_i * g_i), in(f) not in in(G)

1293 // during division).
1294 // c is an integer, and is returned as 'denom'.

1296 // (a) if gcd(c, coeffs(f)) becomes > 1, can we divide
1297 // c, f, by this gcd? If so, how often do we do this?
1298 // (b) do we reduce by any element of the GB, or only those whose
1299 // sugar degree is no greater than degf?
1300 // (c) can we exclude an element of the GB from the g_i?
1301 // (for use in auto reduction).
1302 // (d) can we reduce by the minimal GB instead of the original GB?
